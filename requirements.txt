# Container-side dependencies. Consumers do NOT need these — they only need
# the mineru_client subpackage which has its own pyproject.toml.

# MinerU with vLLM backend support. Pinned minor-locked (>=3.4.0,<3.5.0)
# because MinerU has a history of changing the default VLM model name on
# minor-version bumps (3.1→3.2 swapped 2604→2605 and broke us once), and
# the offline runtime cache cannot follow that without a coordinated
# Dockerfile rebuild. The 3.4.x default VLM is MinerU2.5-Pro-2605-1.2B
# (unchanged since 3.2) and the pipeline OCR is now PP-OCRv6 — both live
# inside the two baked HF repos, so a rebuild picks them up with no bake
# change. Bump deliberately: when crossing minors, re-verify the baked
# models against the new library default in the same commit. Requires
# vllm >=0.10.1.1,<0.22.0 — satisfied by the Dockerfile FROM tag.
mineru[core,vllm]>=3.4.0,<3.5.0

# RunPod serverless SDK — provides runpod.serverless.start() and async helpers.
runpod>=1.7.0

# Used by the handler for downloading URL-supplied PDFs.
httpx>=0.27

# Used by the handler when callers request `return: "s3"` to upload the
# output tarball to an S3-compatible bucket and return a presigned URL.
# Only imported lazily inside the s3 code path, so a worker without S3
# env vars never touches it.
boto3>=1.35

# Xet storage backend client used by huggingface_hub for chunk-based
# parallel downloads. Auto-detected when installed; the Dockerfile bake
# step sets HF_XET_HIGH_PERFORMANCE=1 to saturate bandwidth and CPU
# during the model snapshot pull. Replaces the deprecated hf_transfer.
# Runtime stays offline so this is build-time only.
hf-xet>=1.0

# OpenTelemetry stack. Dormant unless OTEL_EXPORTER_OTLP_ENDPOINT is set
# in the worker environment — worker/telemetry.py reads that env var and
# only imports the SDK / configures exporters when it is non-empty. So
# these packages add ~30 MB to the image but cost zero overhead at
# runtime for the default no-telemetry path. See docs/.../observability.mdx.
# Forks that never want OTel can drop these four lines without other code
# changes.
opentelemetry-api>=1.27,<2
opentelemetry-sdk>=1.27,<2
opentelemetry-exporter-otlp-proto-http>=1.27,<2
# nvidia-ml-py is the upstream-maintained successor to pynvml; we keep
# the `import pynvml` call shape since both packages register the same
# top-level module. Pinned explicitly here so the Dockerfile install
# layer doesn't depend on vllm's transitive pull. Powers the
# mineru.gpu.* gauges in worker/telemetry.py.
nvidia-ml-py>=12
